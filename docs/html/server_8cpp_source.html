14bool isPortBound(int port) {
15 asio::io_service ioService;
16 asio::ip::udp::socket socket_(ioService);
17
18 asio::error_code ec;
19 socket_.open(asio::ip::udp::v4(), ec);
20 if (ec) {
21 std::cout << "Error opening socket: " << ec.message() << std::endl;
22 return false;
23 }
24
25 try {
26 socket_.bind(asio::ip::udp::endpoint(asio::ip::udp::v4(), port), ec);
27 if (ec) {
28 socket_.close(ec);
29 return true; // port is already bound
30 }
31 } catch (const std::exception& ex) {
32 std::cout << "Error binding socket: " << ex.what() << std::endl;
33 return false; // failed to bind port
34 }
35
36 socket_.close(ec);
37 return false; // port is not bound
38}
