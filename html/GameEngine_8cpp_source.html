34void GameEngine::start()
35{
36 while (canStart == false) {
37 _commands.push_back("START\n");
38 #ifdef _WIN32
39 Sleep(1);
40 #else
41 sleep(1);
42 #endif
43 }
44 for (int i = 0; i < _nbPlayers; i++) {
45 EntityID entity = _ecs.createEntity(false);
46 debug("Player " + std::to_string(i));
47 std::string player = "Player " + std::to_string(i) + ":100:" + std::to_string(100 + i * 50) + ":25:25:50:10:2";
48 _commands.push_back(std::string(player));
49 _ecs.addComponent(entity, ID{i});
50 _ecs.addComponent(entity, Health{50});
51 _ecs.addComponent(entity, Transform{100.0f, 100.0f, 0.0f});
52 _ecs.addComponent(entity, Collider{25, 25, "player"});
53 _ecs.addComponent(entity, Pseudo{"Player " + std::to_string(i)});
54 _ecs.addComponent(entity, Score{0});
55 _ecs.addComponent(entity, SkinID{0});
56 }
57 System system(_ecs);
58 float deltaTime = 0.0f;
59 clock_t begin = clock();
60 wave = 0;
61 waveTics = 0;
62 level = 0;
63 int levelSize = _parser._levels.size();
64 globalTics = 0;
65 int updateTic = 3;
66 std::cout << "Level size: " << levelSize << "type " << type << std::endl;
67 while (true) {
68 if (isStopped)
69 break;
70 if (deltaTime > _globalSpeed) {
71 std::cout << "-----------TIC-----------" << std::endl;
72 globalTics++;
73 waveTics++;
74 if (type == 1)
75 levelMode(levelSize);
76 else if (type == 2)
77 waveMode();
78 if (globalTics % updateTic == 0) {
79 try {
80 SendUpdate();
81 } catch (std::exception &e) {
82 std::cout << "Error: " << e.what() << std::endl;
83 }
84 }
85 deltaTime = 0.0f;
86 begin = clock();
87 try {
88 system.handleCollision();
89 system.handleMovement();
90 destroyEnemie();
91 }
92 catch (std::exception &e) {
93 std::cout << "Error: " << e.what() << std::endl;
94 }
95 std::cout << "-------------------------" << std::endl;
96 }
97 deltaTime = float(clock() - begin) / CLOCKS_PER_SEC;
98 }
99}

106void GameEngine::update(std::string command, int id)
107{
108 if (command == "STARTED\n") {
109 canStart = true;
110 return;
111 }
112 debug("Update from client: " + command);
113 try {
114 std::vector<std::string> commands = _parser.stringToArray(command, ' ');
115 if (commands[0] == "Update") {
116 if (commands[1] == "Player") {
117 updatePlayer(commands[2], id);
118 }
119 }
120 if (commands[0] == "Collision") {
121 std::vector<EntityID> entities = _ecs.getEntitiesWithComponent<Collider>();
122 for (int i = 0; i < entities.size(); i++) {
123 if (_ecs.getComponent<Collider>(entities[i]).tag == "enemy") {
124 if (_ecs.getComponent<ID>(entities[i]).id == std::stoi(commands[1])) {
125 _ecs.getComponent<Health>(entities[i]).health -= 10;
126 }
127 }
128 }
129 }
130 if (commands[0] == "Shoot") {
131 _commands.push_back("Shoot " + commands[1]);
132 }
133 if (commands[0] == "Dead") {
134 _commands.push_back("Dead " + commands[1]);
135 }
136 } catch (std::exception &e) {
137 std::cout << "Error: " << e.what() << std::endl;
138 }

193bool GameEngine::waveMode()
194{
195 if (waveTics >= 5 / _globalDifficulty && countObject("enemy") == 0) {
196 debug("Wave tics: " + std::to_string(waveTics));
197 debug("New wave: " + std::to_string(wave));
198 std::vector<Parser::enemy> waves;
199 for (int i = 0; i < (int)_globalDifficulty; i++) {
200 Parser::enemy enemy = _parser.stringToEnemy(std::to_string(eID) + std::string(":1920:1080:10:10:50:10:2:0:180:"));
201 enemy.health = (rand() % 100) * _globalDifficulty;
202 enemy.damage = (rand() % 2) * _globalDifficulty;
203 enemy.speed = ((rand() % 10) + 10);
204 enemy.y = (rand() % 1080);
205 enemy.fireRate = (rand() % 5) + 1;
206 if (wave % 3 == 0 && wave != 0) {
207 enemy.isBoss = 1;
208 enemy.health *= 10;
209 enemy.damage *= 5;
210 enemy.speed *= 0.5;
211 enemy.sizeX *= 2;
212 enemy.sizeY *= 2;
213 }
214 waves.push_back(enemy);
215 eID++;
216 }
217 waveTics = 0;
218 createWave(waves);
219 wave++;
220 _commands.push_back("Wave " + std::to_string(wave));
221 _globalDifficulty += 0.5;
222 }
223 return true;
224}
