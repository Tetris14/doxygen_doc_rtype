220void UDPServer::send_queue()
221{
222 if (_gameEngine == nullptr)
223 return;
224 std::vector<std::string> _commands = _gameEngine->getCommands();
225 if (_commands.size() == 0)
226 return;
227 //std::cout << "Sending queue to client: " << _commands.size() << "\n";
228 int size = _commands.size();
229 int toDel = 0;
230 for (int i = 0; i < size; i++) {
231 sendToAll(_commands[i]);
232 toDel++;
233 }
234 _commands.erase(_commands.begin() + toDel);
235 for (int i = 0; i < size; i++) {
236 _gameEngine->eraseCommands();
237 }
