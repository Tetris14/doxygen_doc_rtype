316void GameScene::checkEnemiesCollision(System &system) {
317 std::vector<EntityID> enemyEntities;
318 std::vector<EntityID> shootEntities;
319 std::vector<EntityID> entities = ecs->getEntitiesWithComponent<Transform>();
320 for (auto &entity : entities) {
321 if (ecs->getComponent<Collider>(entity).tag == "enemy")
322 enemyEntities.push_back(entity);
323 if (ecs->getComponent<Collider>(entity).tag == "shoot")
324 shootEntities.push_back(entity);
325 }
326 for (auto &enemyEntity : enemyEntities) {
327 for (auto &shootEntity : shootEntities) {
328 if (system.CheckCollision(ecs->getComponent<Transform>(enemyEntity).x,
329 ecs->getComponent<Transform>(enemyEntity).y,
330 ecs->getComponent<Collider>(enemyEntity).x,
331 ecs->getComponent<Collider>(enemyEntity).y,
332 ecs->getComponent<Transform>(shootEntity).x,
333 ecs->getComponent<Transform>(shootEntity).y,
334 ecs->getComponent<Collider>(shootEntity).x,
335 ecs->getComponent<Collider>(shootEntity).y)) {
336 std::string collision = "Collision " + std::to_string(ecs->getComponent<ID>(enemyEntity).id);
337 _client->UDPSend(collision + "\n");
338 }
339 }
340 }
